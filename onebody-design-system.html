Section 11

CSS Implementation

Paste these snippets into globals.css. The @theme block registers tokens as Tailwind v4 utilities.

/* globals.css — Tailwind v4 @theme block */
@theme {
  /* Brand colours → text-brand-500, bg-brand-500 etc. */
  --color-brand-500:       #FF33CC;
  --color-brand-600:       #FF33CC;
  --color-brand-700:       #e020b8;
  --color-brand-orange:    #FF6600;

  /* Pastels → bg-pastel-peach, text-pastel-pink etc. */
  --color-pastel-peach:    #FFE8D6;
  --color-pastel-pink:     #FFE0F0;
  --color-pastel-lavender: #F5E6FF;
  --color-pastel-cream:    #FFF8F2;
  --color-pastel-mint:     #E8F8F0;
  --color-pastel-sky:      #E6F2FF;
  --color-pastel-lemon:    #FFF9E0;

  /* Fonts → font-heading, font-body utilities */
  --font-family-heading:   'Montserrat', system-ui, sans-serif;
  --font-family-body:      'Montserrat', system-ui, sans-serif;
}

:root {
  --ob-magenta:      #FF33CC;
  --ob-orange:       #FF6600;
  --ob-gradient:     linear-gradient(90deg, #FF33CC 0%, #FF6600 100%);
  --ob-gradient-135: linear-gradient(135deg, #FF33CC 0%, #FF6600 100%);
  /* Montserrat only — no serif fonts */
  --ob-font-heading: 'Montserrat', system-ui, sans-serif;
  --ob-font-body:    'Montserrat', system-ui, sans-serif;
}

.ob-gradient-text {
  background: var(--ob-gradient);
  -webkit-background-clip: text;
  -webkit-text-fill-color: transparent;
  background-clip: text;
}

.ob-kicker {
  font-family: var(--ob-font-body);
  font-size: 0.7rem;
  font-weight: 500;
  letter-spacing: 0.28em;
  text-transform: uppercase;
  background: var(--ob-gradient);
  -webkit-background-clip: text;
  -webkit-text-fill-color: transparent;
  background-clip: text;
}

.ob-display {
  font-family: var(--ob-font-heading);
  font-size: clamp(3rem, 7.5vw, 8rem);
  font-weight: 700;
  line-height: 0.95;
  letter-spacing: -0.03em;
}
/* layout.tsx — Montserrat only, EB Garamond removed */
<link
  href="https://fonts.googleapis.com/css2?family=Montserrat:ital,wght@0,300;0,400;0,500;0,600;0,700;1,400&display=swap"
  rel="stylesheet"
/>
/* Primary button — gradient, never red */
.btn-primary {
  background: linear-gradient(90deg, #FF33CC 0%, #FF6600 100%);
  color: #fff;
  border-radius: 999px;
  padding: 12px 28px;
  font-weight: 600;
  box-shadow: 0 8px 32px 0 rgba(255,51,204,0.22);
  border: none;
}

/* Focus ring — brand magenta */
.form-input:focus {
  border-color: #FF33CC;
  box-shadow: 0 0 0 3px rgba(255,51,204,0.12);
  outline: none;
}
